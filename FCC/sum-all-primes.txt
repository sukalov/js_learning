function sumPrimes(num) {
  let primes = [];
  for (let i=num; i > 0; i--) {
    let isPrime = true
    for (let j=Math.ceil(i/2); j>1; j--) {
      console.log(i, j , i%j)
      if (i%j == 0) isPrime = false
    }
    if (isPrime) primes.push(i)
  }
  return primes.reduce((sum, prime) => sum + prime) - 1
}
